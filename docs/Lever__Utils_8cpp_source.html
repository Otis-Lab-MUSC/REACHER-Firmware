44 *
45 * Determines whether a lever press is "ACTIVE", "TIMEOUT", or "INACTIVE" based on timing and device states,

53 */
54void definePressActivity(bool programRunning, Lever*& lever, Cue* cue, Pump* pump, Laser* laser) {
55 int32_t timestamp = static_cast<int32_t>(millis()); // Capture initial timestamp
56 if ((cue && cue->isArmed()) && (!pump || !pump->isArmed())) {
57 if (timestamp >= cue->getOnTimestamp() && timestamp <= cue->getOffTimestamp() ||
58 timestamp >= timeoutIntervalStart && timestamp <= timeoutIntervalEnd) {
59 lever->setPressType("TIMEOUT");
60 } else {
61 lever->setPressType("ACTIVE");
62 if (pressCount == fRatio - 1) {
63 pressCount = 0;
64 deliverReward(activeLever, cue, pump, laser);
65 if (programRunning) {
66 timeoutIntervalStart = cue->getOffTimestamp();
67 timeoutIntervalEnd = timeoutIntervalStart + timeoutIntervalLength;
68 }
69 } else {
70 pressCount++;
71 }
72 }
73 } else if ((cue && cue->isArmed()) && (pump && pump->isArmed())) {
74 if (timestamp >= cue->getOnTimestamp() && timestamp <= pump->getInfusionEndTimestamp() ||
75 timestamp >= timeoutIntervalStart && timestamp <= timeoutIntervalEnd) {

101 * @brief Monitors lever pressing with debouncing and triggers associated actions.
102 *
103 * Continuously checks the lever state, applies debouncing, and handles press/release events.
104 *
105 * @param programRunning Boolean indicating if the program is running.
106 * @param lever Reference to a pointer to the Lever object being monitored.
107 * @param cue Pointer to the Cue object (optional, can be nullptr).
108 * @param pump Pointer to the Pump object (optional, can be nullptr).
109 * @param laser Pointer to the Laser object (optional, can be nullptr).
110 */
111void monitorPressing(bool programRunning, Lever*& lever, Cue* cue, Pump* pump, Laser* laser) {
112 static uint32_t lastDebounceTime = 0; // Last time the lever input was toggled
113 const uint32_t debounceDelay = 100; // Debounce time in milliseconds
114 manageCue(cue); // Manage cue delivery
115 managePump(pump); // Manage infusion delivery
116 if (lever->isArmed()) {
117 bool currentLeverState = digitalRead(lever->getPin()); // Read current state
118 if (currentLeverState != lever->getPreviousLeverState()) {
119 lastDebounceTime = millis(); // Reset debouncing timer
120 }
121 if ((millis() - lastDebounceTime) > debounceDelay) {
122 if (currentLeverState != lever->getStableLeverState()) {
123 lever->setStableLeverState(currentLeverState); // Update stable state
124 if (currentLeverState == LOW) { // Lever press detected
125 lever->setPressTimestamp(millis());
126 definePressActivity(programRunning, lever, cue, pump, laser);
127 } else { // Lever release detected
128 lever->setReleaseTimestamp(millis());
129 pressingDataEntry(lever, pump);
130 }
131 }
132 }
133 lever->setPreviousLeverState(currentLeverState); // Update previous state
134 }
